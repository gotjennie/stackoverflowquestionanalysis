I don't (want) to control\u000a# it, and I want to be open to changes that get made to the class\u000a# by the library provider.\u000aclass Programmer(object):\u000a    def __init__(self,name):\u000a        self._name = name\u000a\u000a    def greet(self):\u000a        print "Hi, my name is %s." % self._name\u000a\u000a    def hard_work(self):\u000a        print "The garbage collector will take care of everything."\u000a\u000a# This is my subclass.\u000aclass C_Programmer(Programmer):\u000a    def __init__(self, *args, **kwargs):\u000a        super(C_Programmer,self).__init__(*args, **kwargs)\u000a        self.learn_C()\u000a\u000a    def learn_C(self):\u000a        self._knowledge = ["malloc","free","pointer arithmetic","curly braces"]\u000a\u000a    def hard_work(self):\u000a        print "I'll have to remember " + " and ".join(self._knowledge) + "."\u000a\u000a    # The questionable thing: Reclassing a programmer.\u000a    @classmethod\u000a    def teach_C(cls, programmer):\u000a        programmer.__class__ = cls # &lt;-- do I really want to do this?\u000a        programmer.learn_C()\u000a\u000a\u000ajoel = C_Programmer("Joel")\u000ajoel.greet()\u000ajoel.hard_work()\u000a#&gt;Hi, my name is Joel.\u000a#&gt;I'll have to remember malloc and free and pointer arithmetic and curly braces.\u000a\u000ajeff = Programmer("Jeff")\u000a\u000a# We (or someone else) makes changes to the instance.
